const s = [23, 65, 98, 5];

Array.prototype.myMap = function(callback) {
  const newArray = [];
  // Only change code below this line
  for (let k=0; k<this.length; k++) {
    newArray.push(callback(this[k]));
  }

  // Only change code above this line
  return newArray;
};
